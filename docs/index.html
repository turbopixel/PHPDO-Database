PHPDO

A lightweight PHP PDO database wrapper class.

Version: 1.0.21 | composer require turbopixel/phpdo-database

public connect()

Create database connection.

/**
* @param string $host     Hostname
* @param string $database Database name
* @param string $user     Database user
* @param string $password Database password
* @param int $port        MySQL port
*/


<?php
$PHPDO = new PHPDO();
$PHPDO->connect("database-server.com", "database_name", "user_name", "myPassword123");


public getPdo()

Returns the PHP PDO object.

/**
 * @return PDO
 *
 * @throws \Exception
 */


$PHPDO->getPdo();


public query()

Create pdo object from query.

/**
 * @param string $query
 *
 * @return PDOStatement
 *
 * @throws \Exception
 */


$PHPDO->query("SELECT id FROM user WHERE active = 1");

public prepare()

Run prepared statement.

/**
* @param string $query     MySQL Query
* @param array  $mapping   Data mapping
*
* @return PDOStatement
 *
 * @throws \Exception
*/


$PHPDO->prepare("SELECT id FROM user WHERE id = :userid", ["userid" => 553]);


public execute()

Run raw mysql statement.

/**
* @param string $query     MySQL Query
*
* @return int
 *
 * @throws \Exception
*/


$PHPDO->execute("UPDATE user SET active = 0 WHERE mail IS NULL");


public getLog()

Get query logs. Attribute $PHPDO->logging must be true!

/**
* @return array
*/


$PHPDO->getLog();


public getLastQuery()

Get last query. Attribute $PHPDO->logging must be true!

/**
* @return string
*/


$PHPDO->getLastQuery();
